[DATA TYPE]
INT         : 정수
SERIAL      : 시퀀스 숫자
TEXT        : 문자열
BOOL        : True OR False
DATE        : 날짜
TIMESTAMPTZ : 타임존이 포함된 날짜 및 시간 (만약 현재시간이 필요하면 now())
JSON, 함수 등

[CREATE TABLE]
CREATE TABLE person (
    id SERIAL NOT NULL PRIMARY KEY,
    name TEXT NOT NULL,
    gender TEXT NOT NULL,
    date_of_birth DATE NOT NULL,
    email TEXT,
    create_at TIMESTAMPTZ DEFAULT now()
);
